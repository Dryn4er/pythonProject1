from typing_extensions import Union

account_number = input()
card_number = input()


def get_mask_card_number(card_number: Union[int, str]) -> Union[str]:
    """Функция, которая маскирует номер карты"""
    card_mask = [card_number]
    return f"{card_mask[:4]}{card_mask[4:6]}** **** {card_mask[12:]}"


def get_mask_account(account_number: Union[int, str]) -> Union[str]:
    """Функция, которая маскирует номер счета"""
    account_mask = [account_number]
    return f"**{account_mask[-4:]}"


print(get_mask_card_number(card_number))
print(get_mask_account(account_number))
